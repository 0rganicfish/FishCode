1. 选择排序

   ```C++
   void pai(int a[], int n)
   {
       for (int i = 0; i < n - 1; i++)
       {
           int minn = i;
           for (int j = i + 1; j < n; j++)
               if (a[minn] > a[j])
                   minn = j;
           swap(a[i], a[minn]);
       }
   }
   ```
